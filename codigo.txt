#include <iostream>

class cuentaObjeto {
private:
    static int objCreados;       // Declarar como estático para contar objetos en toda la clase
    static int objDestruidos;    // Declarar como estático para contar objetos en toda la clase

public:
    cuentaObjeto() {
        objCreados++;
    }

    cuentaObjeto(const cuentaObjeto&) {
        objCreados++;
    }

    cuentaObjeto& metodo() {
        return *this;
    }

    static void mostrarResumen() {
        std::cout << "Objetos creados: " << objCreados << std::endl;
        std::cout << "Objetos destruidos: " << objDestruidos << std::endl;
    }

    ~cuentaObjeto() {
        objDestruidos++;
    }
};

// Inicializar los contadores estáticos
int cuentaObjeto::objCreados = 0;
int cuentaObjeto::objDestruidos = 0;

int main() {
    std::cout << "!!!Hello World!!!" << std::endl;

    cuentaObjeto obj1;
    cuentaObjeto::mostrarResumen(); // Mostrar el resumen después de la primera creación

    {
        cuentaObjeto obj2(obj1);
        cuentaObjeto::mostrarResumen(); // Mostrar el resumen después de la segunda creación

        cuentaObjeto *obj3 = new cuentaObjeto();
        *obj3 = obj2.metodo();
        cuentaObjeto::mostrarResumen(); // Mostrar el resumen después de la tercera creación

        delete obj3; // Liberar la memoria del objeto creado dinámicamente
    }

    cuentaObjeto::mostrarResumen(); // Mostrar el resumen final

    return 0;
}
